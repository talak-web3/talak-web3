STEP 1: Fix In-Memory Store Memory Leaks (SEC-001 / BUG-001)
  File: packages/auth/src/index.ts
  Issue: InMemoryNonceStore, InMemoryRefreshStore, InMemoryRevocationStore use
         plain Maps with no TTL eviction. Memory grows unboundedly.
  Fix:
    a) Add a private `startCleanup()` method to each InMemory store class that
       runs `setInterval` every 60 seconds to sweep expired entries.
    b) Add a public `destroy()` method that calls `clearInterval`.
    c) For InMemoryNonceStore: sweep entries where `Date.now() > entry.expiresAt`.
    d) For InMemoryRefreshStore: sweep entries where `Date.now() > session.expiresAt`.
    e) For InMemoryRevocationStore: sweep entries where `Date.now() > entry.expiresAt`.
    f) Call `startCleanup()` in each constructor.

  Exact implementation for InMemoryNonceStore (apply same pattern to others):

    private cleanupInterval: ReturnType<typeof setInterval> | null = null;

    constructor() {
      this.startCleanup();
    }

    private startCleanup(): void {
      this.cleanupInterval = setInterval(() => {
        const now = Date.now();
        for (const [key, entry] of this.store.entries()) {
          if (now > entry.expiresAt) {
            this.store.delete(key);
          }
        }
      }, 60_000);
      // Allow process to exit even if interval is running
      if (this.cleanupInterval.unref) {
        this.cleanupInterval.unref();
      }
    }

    destroy(): void {
      if (this.cleanupInterval) {
        clearInterval(this.cleanupInterval);
        this.cleanupInterval = null;
      }
    }

  Also export the `destroy()` method from the package so consumers can call it.

  File: packages/core/src/security.ts

  Issue: Only matches `/0x[a-fA-F0-9]{64}/g`. Misses PEM keys, mnemonics, base64.
  Fix: Replace the checkSecrets method with comprehensive detection:

    static checkSecrets(config: unknown): void {
      if (typeof config !== "object" || config === null) return;
      const configStr = JSON.stringify(config);

      // 1. 0x-prefixed hex private keys (existing check)
      if (/0x[0-9a-fA-F]{64}/.test(configStr)) {
        throw new TalakWeb3Error("Potential private key leak detected in config (hex)", {
          code: SECURITY_ERROR_CODES.SECRET_LEAK, status: 400,
        });
      }

      // 2. PEM-encoded private keys
      if (/-----BEGIN (RSA |EC )?PRIVATE KEY-----/.test(configStr)) {
        throw new TalakWeb3Error("Potential private key leak detected in config (PEM)", {
          code: SECURITY_ERROR_CODES.SECRET_LEAK, status: 400,
        });
      }

      // 3. Raw hex strings of 64+ chars (no 0x prefix)
      if (/(?:^|[^0-9a-fA-F])([0-9a-fA-F]{64})(?:[^0-9a-fA-F]|$)/.test(configStr)) {
        throw new TalakWeb3Error("Potential secret leak detected in config (raw hex)", {
          code: SECURITY_ERROR_CODES.SECRET_LEAK, status: 400,
        });

      // 4. Base64-encoded keys (44+ chars that look like base64)
      if (/(?:^|[\s":])([A-Za-z0-9+/]{43}=)(?:[\s",}]|$)/.test(configStr)) {
        throw new TalakWeb3Error("Potential secret leak detected in config (base64)", {
          code: SECURITY_ERROR_CODES.SECRET_LEAK, status: 400,
        });
      }
    }

STEP 4: Fix Prototype Pollution Check (SEC-004)
  File: packages/core/src/security.ts
  Lines: 33-38
  Issue: Serializes to JSON then checks string — bypassable at object level.
  Fix: Replace validateRpcParams with deep object scanning:

    static validateRpcParams(params: unknown[]): void {
      function hasDangerousKeys(obj: unknown, depth = 0): boolean {
        if (depth > 10) return false;
        if (obj === null || typeof obj !== "object") return false;
        if (Array.isArray(obj)) {
          return obj.some((item) => hasDangerousKeys(item, depth + 1));
        }
        for (const key of Object.keys(obj as Record<string, unknown>)) {
          if (key === "__proto__" || key === "constructor" || key === "prototype") {
            return true;
          }
          if (hasDangerousKeys((obj as Record<string, unknown>)[key], depth + 1)) {
            return true;
          }
        }
        return false;
      }

      if (hasDangerousKeys(params)) {
        throw new TalakWeb3Error("Disallowed parameter key detected", {
          code: SECURITY_ERROR_CODES.INVALID_PARAM, status: 400,
        });
      }
    }
